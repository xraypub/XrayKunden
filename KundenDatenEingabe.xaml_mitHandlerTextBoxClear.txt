namespace XrayKunden.Views
{
    /// <summary>
    /// Interaktionslogik für KundenDatenEingabe.xaml
    /// </summary>
    public partial class KundenDatenEingabe : UserControl
    {
        public KundenDatenEingabe()
        {
            InitializeComponent();
        }

        private void TextBox_GotFocus(object sender, RoutedEventArgs e)
        {
            var textbox = sender as TextBox;

            switch (textbox?.Text)
            {
                case  "Alle Vornamen eingeben":
                     textbox?.Clear();
                    break;

                case "Familienname eingeben":
                    textbox?.Clear();
                    break;

                case "Strasse und Hausnummer eingeben":
                    textbox?.Clear();
                    break;

                case "Ort eingeben":
                    textbox?.Clear();
                    break;

                case "Postleitzahl eingeben":
                    textbox?.Clear();
                    break;

                case "Kunden-Nummer vergeben":
                    textbox?.Clear();
                    break;
                    
                    default: break;

            }

            
        }

       
    }
}
